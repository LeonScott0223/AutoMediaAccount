# 数据表设计

## 说明

- 当前 C 端没有后端 Model/Entity，以下表结构根据 `web/assets/js/data.js` 与 C 端页面流程反推生成。
- SQL 兼容 MySQL 5.7+，统一使用 `utf8mb4`。
- 金额字段使用 `decimal(10,2)`，时间字段使用 `datetime`，状态字段使用 `tinyint(1)` 或 `varchar`。

```sql
CREATE TABLE `aa_admin_users` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  `username` varchar(50) NOT NULL DEFAULT '' COMMENT '管理员用户名',
  `password_hash` varchar(255) NOT NULL DEFAULT '' COMMENT '密码哈希',
  `nickname` varchar(50) NOT NULL DEFAULT '' COMMENT '管理员昵称',
  `status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '状态：1正常 0禁用',
  `last_login_at` datetime DEFAULT NULL COMMENT '最近登录时间',
  `remark` varchar(500) NOT NULL DEFAULT '' COMMENT '备注',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_username` (`username`),
  KEY `idx_status` (`status`),
  KEY `idx_created_at` (`created_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='后台管理员表';

CREATE TABLE `aa_users` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  `username` varchar(50) NOT NULL DEFAULT '' COMMENT '用户名',
  `password_hash` varchar(255) NOT NULL DEFAULT '' COMMENT '密码哈希',
  `nickname` varchar(50) NOT NULL DEFAULT '' COMMENT '昵称',
  `status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '状态：1正常 0禁用',
  `last_login_at` datetime DEFAULT NULL COMMENT '最近登录时间',
  `remark` varchar(500) NOT NULL DEFAULT '' COMMENT '备注',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '注册时间',
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_username` (`username`),
  KEY `idx_status` (`status`),
  KEY `idx_created_at` (`created_at`),
  KEY `idx_last_login_at` (`last_login_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='C端用户表';

CREATE TABLE `aa_account_platforms` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  `name` varchar(30) NOT NULL DEFAULT '' COMMENT '账号分类名称',
  `code` varchar(30) NOT NULL DEFAULT '' COMMENT '账号分类编码',
  `status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '状态：1启用 0禁用',
  `sort_order` int(11) NOT NULL DEFAULT '0' COMMENT '排序值',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_code` (`code`),
  KEY `idx_status_sort` (`status`,`sort_order`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='账号分类表';

CREATE TABLE `aa_audience_categories` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  `name` varchar(30) NOT NULL DEFAULT '' COMMENT '受众偏好名称',
  `code` varchar(30) NOT NULL DEFAULT '' COMMENT '受众偏好编码',
  `status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '状态：1启用 0禁用',
  `sort_order` int(11) NOT NULL DEFAULT '0' COMMENT '排序值',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_code` (`code`),
  KEY `idx_status_sort` (`status`,`sort_order`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='受众偏好表';

CREATE TABLE `aa_products` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  `product_code` varchar(50) NOT NULL DEFAULT '' COMMENT '商品编号',
  `platform_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '账号分类ID',
  `audience_category_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '受众偏好ID',
  `title` varchar(120) NOT NULL DEFAULT '' COMMENT '商品标题',
  `subtitle` varchar(255) NOT NULL DEFAULT '' COMMENT '商品副标题',
  `price` decimal(10,2) NOT NULL DEFAULT '0.00' COMMENT '商品价格',
  `image_url` varchar(255) NOT NULL DEFAULT '' COMMENT '商品图片',
  `fans_text` varchar(20) NOT NULL DEFAULT '' COMMENT '粉丝展示值',
  `fans_value` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '粉丝数值',
  `favorite_text` varchar(20) NOT NULL DEFAULT '' COMMENT '收藏展示值',
  `favorite_value` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '收藏数值',
  `gender_tag` varchar(30) NOT NULL DEFAULT '' COMMENT '性别标签',
  `gender_ratio` varchar(80) NOT NULL DEFAULT '' COMMENT '男女粉比例',
  `fans_profile` varchar(500) NOT NULL DEFAULT '' COMMENT '粉丝画像',
  `is_certified` tinyint(1) unsigned NOT NULL DEFAULT '0' COMMENT '是否已实名：1是 0否',
  `certified_text` varchar(20) NOT NULL DEFAULT '' COMMENT '实名展示文案',
  `certification_mode` varchar(30) NOT NULL DEFAULT '' COMMENT '实名模式：固定实名/可自定义实名',
  `introduction` text NOT NULL COMMENT '商品介绍',
  `instructions` text NOT NULL COMMENT '使用说明',
  `usage_notes` text NOT NULL COMMENT '常见说明JSON',
  `stock_total` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '总库存',
  `stock_available` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '可售库存',
  `sales_count` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '销量',
  `status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '商品状态：1上架 0下架',
  `sort_order` int(11) NOT NULL DEFAULT '0' COMMENT '排序值',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_product_code` (`product_code`),
  KEY `idx_platform_id` (`platform_id`),
  KEY `idx_audience_category_id` (`audience_category_id`),
  KEY `idx_price` (`price`),
  KEY `idx_fans_value` (`fans_value`),
  KEY `idx_favorite_value` (`favorite_value`),
  KEY `idx_certified_status` (`is_certified`,`status`),
  KEY `idx_status_sort` (`status`,`sort_order`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='商品表';

CREATE TABLE `aa_product_cards` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  `product_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '商品ID',
  `account_value` varchar(100) NOT NULL DEFAULT '' COMMENT '账号',
  `password_value` varchar(100) NOT NULL DEFAULT '' COMMENT '密码',
  `card_status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '状态：1未售出 2已锁定 3已售出 0已禁用',
  `locked_order_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '锁定订单ID',
  `sold_order_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '售出订单ID',
  `locked_at` datetime DEFAULT NULL COMMENT '锁定时间',
  `sold_at` datetime DEFAULT NULL COMMENT '售出时间',
  `remark` varchar(500) NOT NULL DEFAULT '' COMMENT '备注',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '导入时间',
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  KEY `idx_product_status` (`product_id`,`card_status`),
  KEY `idx_account_value` (`account_value`),
  KEY `idx_locked_order_id` (`locked_order_id`),
  KEY `idx_sold_order_id` (`sold_order_id`),
  KEY `idx_sold_at` (`sold_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='卡密库存表';

CREATE TABLE `aa_card_import_batches` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  `batch_no` varchar(64) NOT NULL DEFAULT '' COMMENT '导入批次号',
  `product_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '商品ID',
  `total_count` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '导入总数',
  `success_count` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '成功数量',
  `failed_count` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '失败数量',
  `import_status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '导入状态：1成功 2部分成功 0失败',
  `import_format` varchar(30) NOT NULL DEFAULT '' COMMENT '导入格式',
  `overwrite_duplicate` tinyint(1) unsigned NOT NULL DEFAULT '0' COMMENT '是否覆盖重复账号：1是 0否',
  `failed_file_url` varchar(255) NOT NULL DEFAULT '' COMMENT '失败记录文件地址',
  `operator_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '操作人ID',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '导入时间',
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_batch_no` (`batch_no`),
  KEY `idx_product_id` (`product_id`),
  KEY `idx_import_status` (`import_status`),
  KEY `idx_created_at` (`created_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='卡密导入批次表';

CREATE TABLE `aa_payment_methods` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  `method_name` varchar(30) NOT NULL DEFAULT '' COMMENT '支付方式名称',
  `method_code` varchar(30) NOT NULL DEFAULT '' COMMENT '支付方式编码',
  `qr_prefix` varchar(255) NOT NULL DEFAULT '' COMMENT '二维码前缀',
  `status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '状态：1启用 0禁用',
  `sort_order` int(11) NOT NULL DEFAULT '0' COMMENT '排序值',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_method_code` (`method_code`),
  KEY `idx_status_sort` (`status`,`sort_order`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='支付方式表';

CREATE TABLE `aa_orders` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  `order_no` varchar(32) NOT NULL DEFAULT '' COMMENT '订单号',
  `user_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '用户ID，免登录订单为0',
  `product_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '商品ID',
  `product_title` varchar(120) NOT NULL DEFAULT '' COMMENT '商品标题快照',
  `buyer_name` varchar(50) NOT NULL DEFAULT '' COMMENT '联系人',
  `contact_value` varchar(100) NOT NULL DEFAULT '' COMMENT '联系方式',
  `remarks` varchar(500) NOT NULL DEFAULT '' COMMENT '备注说明',
  `pay_method_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '支付方式ID',
  `pay_method_name` varchar(30) NOT NULL DEFAULT '' COMMENT '支付方式名称快照',
  `order_amount` decimal(10,2) NOT NULL DEFAULT '0.00' COMMENT '订单金额',
  `order_status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '订单状态：1待支付 2已支付 3已取消 4已过期',
  `delivery_status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '交付状态：1待支付 2待发货 3已发货 4发货失败',
  `card_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '交付卡密ID',
  `account_value` varchar(100) NOT NULL DEFAULT '' COMMENT '交付账号快照',
  `password_value` varchar(100) NOT NULL DEFAULT '' COMMENT '交付密码快照',
  `payment_expire_at` datetime DEFAULT NULL COMMENT '支付过期时间',
  `paid_at` datetime DEFAULT NULL COMMENT '支付时间',
  `delivered_at` datetime DEFAULT NULL COMMENT '交付时间',
  `aftersale` varchar(500) NOT NULL DEFAULT '' COMMENT '售后说明',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '下单时间',
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_order_no` (`order_no`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_product_id` (`product_id`),
  KEY `idx_contact_value` (`contact_value`),
  KEY `idx_order_status` (`order_status`),
  KEY `idx_delivery_status` (`delivery_status`),
  KEY `idx_created_at` (`created_at`),
  KEY `idx_paid_at` (`paid_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='订单表';

CREATE TABLE `aa_order_timelines` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  `order_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '订单ID',
  `timeline_text` varchar(120) NOT NULL DEFAULT '' COMMENT '流程文案',
  `sort_order` int(11) NOT NULL DEFAULT '0' COMMENT '排序值',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `idx_order_sort` (`order_id`,`sort_order`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='订单流程表';

CREATE TABLE `aa_payment_records` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  `payment_no` varchar(64) NOT NULL DEFAULT '' COMMENT '支付流水号',
  `order_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '订单ID',
  `order_no` varchar(32) NOT NULL DEFAULT '' COMMENT '订单号',
  `pay_method_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '支付方式ID',
  `pay_method_name` varchar(30) NOT NULL DEFAULT '' COMMENT '支付方式名称',
  `pay_amount` decimal(10,2) NOT NULL DEFAULT '0.00' COMMENT '支付金额',
  `pay_status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '支付状态：1待支付 2支付成功 3支付失败 4已过期',
  `qr_code_url` varchar(500) NOT NULL DEFAULT '' COMMENT '二维码地址',
  `expire_at` datetime NOT NULL COMMENT '过期时间',
  `paid_at` datetime DEFAULT NULL COMMENT '支付时间',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_payment_no` (`payment_no`),
  KEY `idx_order_id` (`order_id`),
  KEY `idx_order_no` (`order_no`),
  KEY `idx_pay_status` (`pay_status`),
  KEY `idx_expire_at` (`expire_at`),
  KEY `idx_paid_at` (`paid_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='支付记录表';

CREATE TABLE `aa_notices` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  `title` varchar(80) NOT NULL DEFAULT '' COMMENT '公告标题',
  `content` varchar(1000) NOT NULL DEFAULT '' COMMENT '公告内容',
  `time_label` varchar(30) NOT NULL DEFAULT '' COMMENT '时间标签',
  `status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '公告状态：1显示 0隐藏',
  `auto_popup` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '是否自动弹窗：1是 0否',
  `sort_order` int(11) NOT NULL DEFAULT '0' COMMENT '排序值',
  `published_at` datetime DEFAULT NULL COMMENT '发布时间',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  KEY `idx_status_popup_sort` (`status`,`auto_popup`,`sort_order`),
  KEY `idx_published_at` (`published_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='网站公告表';

CREATE TABLE `aa_site_configs` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  `config_key` varchar(80) NOT NULL DEFAULT '' COMMENT '配置键',
  `config_name` varchar(80) NOT NULL DEFAULT '' COMMENT '配置名称',
  `config_value` text NOT NULL COMMENT '配置值',
  `status` tinyint(1) unsigned NOT NULL DEFAULT '1' COMMENT '状态：1启用 0禁用',
  `remark` varchar(500) NOT NULL DEFAULT '' COMMENT '备注',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_config_key` (`config_key`),
  KEY `idx_status` (`status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='站点配置表';
```
